Note: All endpoints require a Bearer token. Authenticate first using POST /api/auth/login, then pass the token as Authorization: Bearer <token>.
POST /api/auth/login Get auth token 
curl -s -X POST http://localhost:3000/api/auth/login \
  -H "Content-Type: application/json" \
  -d '{"username":"admin","password":"Ideal@2024"}'
POST /api/test/generate Generate random test orders 
curl -s -X POST http://localhost:3000/api/test/generate \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer YOUR_TOKEN" \
  -d '{
    "count": 5,
    "clientId": "betime-marketing",
    "channel": "shopee",
    "status": "pending"
  }'
POST /api/test/inject Inject custom order objects 
curl -s -X POST http://localhost:3000/api/test/inject \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer YOUR_TOKEN" \
  -d '{
    "orders": [
      {
        "id": "MY-TEST-001",
        "clientId": "betime-marketing",
        "clientName": "Betime Marketing",
        "channel": "shopee",
        "orderDate": "2026-07-03T10:00:00Z",
        "status": "pending",
        "currency": "SGD",
        "notes": "Test order",
        "items": [
          { "sku": "SKU-001", "name": "Test Product", "qty": 2, "unitPrice": 25.00 }
        ],
        "shipping": {
          "recipient": "John Doe",
          "addressLine1": "123 Test Street",
          "city": "Singapore",
          "zip": "520123",
          "country": "SG"
        },
        "subtotal": 50.00,
        "shippingCost": 0,
        "tax": 4.50,
        "total": 54.50
      }
    ]
  }'
POST /api/test/inject Inject from a JSON file 
curl -s -X POST http://localhost:3000/api/test/inject \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer YOUR_TOKEN" \
  -d @my_orders.json
DELETE /api/test/orders Remove all test orders 
curl -s -X DELETE http://localhost:3000/api/test/orders \
  -H "Authorization: Bearer YOUR_TOKEN"
POST /api/test/generate Generate for all clients, all channels 
curl -s -X POST http://localhost:3000/api/test/generate \

  -H "Authorization: Bearer YOUR_TOKEN" \
